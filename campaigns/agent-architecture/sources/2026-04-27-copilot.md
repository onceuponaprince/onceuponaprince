## 1) Predictive features: what actually moves routing quality

The most predictive features for “single-shot vs orchestrated” and “monolithic vs decomposed” are usually:

1. **Surface complexity**: token count, sentence count, code fences, file-path mentions, stacktrace markers.
2. **Operational intent**: verbs like *fix/build/refactor/debug/migrate/analyse* vs *summarise/explain*.
3. **Actionability footprint**: expected tool fan-out (bash+edit+search), multi-file references, explicit acceptance criteria.
4. **Ambiguity/risk**: unresolved pronouns, missing constraints, open-ended asks.
5. **State dependence**: references to prior steps (“continue”, “as above”, “based on previous result”).

A practical extractor (drop-in for routing):

```python
import re
from dataclasses import dataclass, asdict
from typing import Dict, List
from collections import Counter

ACTION_VERBS = {
    "fix","debug","implement","refactor","migrate","write","test","benchmark",
    "deploy","configure","optimize","investigate","compare","route","classify"
}
LOW_COMPLEXITY_VERBS = {"summarize","explain","define","translate","rewrite"}

FILE_PAT = re.compile(r'(?:(?:[\w\-.]+/)+)?[\w\-.]+\.(?:py|ts|tsx|js|go|java|rb|rs|sql|md|yaml|yml|json|toml)')
CODE_FENCE_PAT = re.compile(r"```")
STACKTRACE_PAT = re.compile(r"(Traceback|Exception|Error:|at\s+\w+\s+\()")
LIST_STEP_PAT = re.compile(r"^\s*\d+\.\s", re.M)

@dataclass
class TaskFeatures:
    char_len: int
    token_est: int
    sentence_count: int
    code_fence_count: int
    file_mention_count: int
    stacktrace_hits: int
    numbered_steps: int
    action_verb_hits: int
    low_complexity_verb_hits: int
    ambiguity_score: float
    decomposition_score: float

def simple_token_estimate(text: str) -> int:
    return max(1, int(len(text.split()) * 1.2))

def ambiguity_score(text: str) -> float:
    # crude but useful: unresolved pointers + vague terms
    vague = len(re.findall(r"\b(this|that|it|something|stuff|maybe|etc)\b", text.lower()))
    explicit_constraints = len(re.findall(r"\b(must|exactly|only|do not|never|required)\b", text.lower()))
    return max(0.0, (vague - 0.5 * explicit_constraints) / 10.0)

def extract_features(task_text: str) -> Dict:
    lower = task_text.lower()
    words = re.findall(r"\b[a-z]+\b", lower)
    c = Counter(words)
    action_hits = sum(c[v] for v in ACTION_VERBS)
    low_hits = sum(c[v] for v in LOW_COMPLEXITY_VERBS)

    file_mentions = FILE_PAT.findall(task_text)
    code_fences = len(CODE_FENCE_PAT.findall(task_text))
    stack_hits = len(STACKTRACE_PAT.findall(task_text))
    steps = len(LIST_STEP_PAT.findall(task_text))

    decomp = (
        0.25 * min(1.0, len(file_mentions) / 4)
        + 0.2 * min(1.0, code_fences / 2)
        + 0.2 * min(1.0, steps / 4)
        + 0.2 * min(1.0, action_hits / 6)
        + 0.15 * min(1.0, stack_hits / 2)
    )

    f = TaskFeatures(
        char_len=len(task_text),
        token_est=simple_token_estimate(task_text),
        sentence_count=max(1, len(re.split(r"[.!?]\s+", task_text.strip()))),
        code_fence_count=code_fences,
        file_mention_count=len(file_mentions),
        stacktrace_hits=stack_hits,
        numbered_steps=steps,
        action_verb_hits=action_hits,
        low_complexity_verb_hits=low_hits,
        ambiguity_score=ambiguity_score(task_text),
        decomposition_score=decomp,
    )
    return asdict(f)
```

**Why these work:** they correlate directly with tool-calling load and branching depth (the actual cost drivers in agent orchestration).

## 3) Classifier shape: cheapest options that actually help

### (a) Rule-based heuristics (best weekend v0)

```python
def rule_router(features: dict) -> tuple[str, float]:
    score = 0.0
    score += 0.35 if features["token_est"] > 180 else 0
    score += 0.25 if features["file_mention_count"] >= 2 else 0
    score += 0.15 if features["code_fence_count"] > 0 else 0
    score += 0.15 if features["action_verb_hits"] >= 3 else 0
    score += 0.10 if features["ambiguity_score"] > 0.3 else 0
    label = "orchestrated_decomposed" if score >= 0.45 else "single_shot_monolith"
    return label, score
```

### (b) BERT-class / MiniLM (best cost-quality for solo)

```python
from sentence_transformers import SentenceTransformer
from sklearn.linear_model import LogisticRegression
import numpy as np

embedder = SentenceTransformer("sentence-transformers/all-MiniLM-L6-v2")
X = embedder.encode(train_texts, normalize_embeddings=True)
y = np.array(train_labels)  # 0/1

clf.fit(X, y)

def bert_router(task_text: str):
    v = embedder.encode([task_text], normalize_embeddings=True)
    p = clf.predict_proba(v)[0, 1]

## 4) Measurement loop: A/B, shadow mode, online eval

### LangGraph (branching + conditional edges)

LangGraph `StateGraph` + `add_conditional_edges` is the right primitive for route A/B and fallback chains (see tests using `add_conditional_edges` and `set_conditional_entry_point`)  
Source: [langgraph test_pregel.py @ 53a9806](https://github.com/langchain-ai/langgraph/blob/53a9806e659552f91adf5bb0f2de5820b87a55a0/libs/langgraph/tests/test_pregel.py).

```python
from typing import TypedDict, Literal
from langgraph.graph import StateGraph, START, END

class S(TypedDict):
    task: str
    route: str
    output: str

def classify(s: S): ...
def mono(s: S): ...
def orch(s: S): ...

def route_fn(s: S) -> Literal["mono", "orch"]:
    return "orch" if s["route"] == "orchestrated_decomposed" else "mono"

g = StateGraph(S)
g.add_node("classify", classify)
g.add_node("mono", mono)
g.add_node("orch", orch)
g.add_edge(START, "classify")
g.add_conditional_edges("classify", route_fn)
g.add_edge("mono", END)
g.add_edge("orch", END)
app = g.compile()
```

### OpenAI Agents SDK (handoff filters + RunConfig)

APIs are explicit: `handoff(..., input_filter=...)`, `handoff_filters.remove_all_tools`, `Runner.run(..., run_config=RunConfig(...))`, and global `ModelSettings.tool_choice` knob.  
Sources: [`handoffs/__init__.py`](https://github.com/openai/openai-agents-python/blob/8025ed0b426f7bdc63e489d1bfc1b9af5c3f3f2b/src/agents/handoffs/__init__.py), [`handoff_filters.py`](https://github.com/openai/openai-agents-python/blob/8025ed0b426f7bdc63e489d1bfc1b9af5c3f3f2b/src/agents/extensions/handoff_filters.py), [`run_config.py`](https://github.com/openai/openai-agents-python/blob/8025ed0b426f7bdc63e489d1bfc1b9af5c3f3f2b/src/agents/run_config.py), [`model_settings.py`](https://github.com/openai/openai-agents-python/blob/8025ed0b426f7bdc63e489d1bfc1b9af5c3f3f2b/src/agents/model_settings.py), [`message_filter.py`](https://github.com/openai/openai-agents-python/blob/8025ed0b426f7bdc63e489d1bfc1b9af5c3f3f2b/examples/handoffs/message_filter.py).

from agents import Agent, Runner, handoff

## 5) Failure modes + fallback patterns

Common failure modes:
1. **Over-routing to orchestrated** (cost explosion).
2. **Under-routing** (single-shot fails on multi-step tasks).
3. **Classifier drift** (new task distribution).
4. **Low-confidence ambiguity** (unstable route flips).
5. **Tool-chain deadlocks** (handoff loops, repeated retries).

Concrete middleware pattern:

```python
from dataclasses import dataclass
from time import time

@dataclass
class RouteDecision:
    label: str
    confidence: float
    reason: str

class CircuitBreaker:
    def __init__(self, fail_threshold=5, cooloff_s=120):
        self.fail_threshold = fail_threshold
        self.cooloff_s = cooloff_s
        self.fails = 0
        self.opened_at = None

    def allow(self):
        if self.opened_at is None: return True
        return (time() - self.opened_at) > self.cooloff_s

    def record(self, ok: bool):
        if ok:
            self.fails = 0
            self.opened_at = None
        else:
            self.fails += 1
            if self.fails >= self.fail_threshold:
                self.opened_at = time()

def guarded_route(task: str, classifier, cb: CircuitBreaker) -> RouteDecision:
    if not cb.allow():
        return RouteDecision("single_shot_monolith", 1.0, "circuit_open_fallback")

    label, conf = classifier(task)
    if conf < 0.55:
        return RouteDecision("single_shot_monolith", conf, "low_confidence_fallback")

    return RouteDecision(label, conf, "classifier")

def escalation_chain(task, decision: RouteDecision):
    if decision.label == "orchestrated_decomposed":
        return run_orchestrated(task)
    out = run_monolith(task)
    if not out.success:   # explicit failed outcome, not silent swallow
        return run_orchestrated(task)  # one-step escalation
    return out
```
